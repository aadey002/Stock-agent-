// Load market conditions
        async function loadMarketConditions() {
            try {
                const response = await fetch('data/market_conditions.json');
                if (response.ok) {
                    const data = await response.json();
                    updateMarketConditions(data);
                }
            } catch (e) {
                console.log('Market conditions not available');
            }
        }
        
        function updateMarketConditions(data) {
            const score = data.overall_score || 50;
            const scoreEl = document.getElementById('marketScore');
            const labelEl = document.getElementById('marketLabel');
            const cardEl = document.getElementById('marketConditionsCard');
            
            scoreEl.textContent = score + '/100';
            
            if (score >= 70) {
                scoreEl.className = 'status-value status-healthy';
                labelEl.textContent = 'üü¢ Favorable - Normal trading';
                cardEl.style.borderTop = '4px solid #10b981';
            } else if (score >= 50) {
                scoreEl.className = 'status-value status-warning';
                labelEl.textContent = 'üü° Moderate - Trade with caution';
                cardEl.style.borderTop = '4px solid #f59e0b';
            } else if (score >= 30) {
                scoreEl.className = 'status-value status-warning';
                labelEl.textContent = 'üü† Caution - Reduce exposure';
                cardEl.style.borderTop = '4px solid #f97316';
            } else {
                scoreEl.className = 'status-value status-error';
                labelEl.textContent = 'üî¥ Poor - Consider avoiding trades';
                cardEl.style.borderTop = '4px solid #ef4444';
            }
            
            // Update individual indicators
            const vol = data.conditions?.volatility || {};
            const sent = data.conditions?.sentiment || {};
            const rot = data.conditions?.sector_rotation || {};
            const cal = data.conditions?.economic_calendar || {};
            
            document.getElementById('vixLevel').textContent = vol.level || 'N/A';
            document.getElementById('sentimentLevel').textContent = sent.sentiment || 'N/A';
            document.getElementById('rotationLevel').textContent = rot.rotation || 'N/A';
            document.getElementById('eventsLevel').textContent = (cal.events_today || 0) + ' today';
        }
        
        // Load on page load
        loadMarketConditions();
